mousetrap.js

type some stuff and look over there

// single keys down
Mousetrap.bind('4', function() { debug('4 down'); });
Mousetrap.bind("$", function() { debug('dollar sign down'); }, 'down');
Mousetrap.bind('i', function() { debug('i down'); });

// single keys up
Mousetrap.bind('4', function() { debug('4 up'); }, 'up');
Mousetrap.bind("$", function() { debug('dollar sign up'); }, 'up');

// combinations down
Mousetrap.bind('ctrl + shift + up', function() { debug('control shift up'); });
Mousetrap.bind('ctrl+shift+down', function() { debug('control shift down'); });

// combinations up
Mousetrap.bind('shift + Q', function() { debug('shift q down')});
Mousetrap.bind('shift + q', function() { debug('shift q up'); }, 'up');

// gmail style sequences
Mousetrap.bind('g i', function() { debug('go to inbox'); });
Mousetrap.bind('g t', function() { debug('go to sent'); });

Mousetrap.bind('* a', function() { debug('select all')});
Mousetrap.bind('* n', function() { debug('select none')});

Mousetrap.bind('up up down down left right left right b a enter', function() {
    debug('konami code!');
});

Mousetrap.bind('c r a i g', function() { debug('craig'); });

// combination and sequence together!
Mousetrap.bind('shift+c r a i g', function() { debug('capital craig'); });
