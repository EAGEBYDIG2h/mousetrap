// single keys down
Mousetrap.bind('4', function() { append('4 down'); });
Mousetrap.bind("$", function() { append('dollar sign down'); }, 'down');
Mousetrap.bind('i', function() { append('i down'); });

// single keys up
Mousetrap.bind('4', function() { append('4 up'); }, 'up');
Mousetrap.bind("$", function() { append('dollar sign up'); }, 'up');

// combinations down
Mousetrap.bind('ctrl + shift + up', function() { append('control shift up'); });
Mousetrap.bind('ctrl+shift+down', function() { append('control shift down'); });

// combinations up
Mousetrap.bind('shift + Q', function() { append('shift q down')});
Mousetrap.bind('shift + q', function() { append('shift q up'); }, 'up');

// sequences
Mousetrap.bind('g i', function() { append('go to inbox'); });
Mousetrap.bind('g t', function() { append('go to sent'); });

Mousetrap.bind('up up down down left right left right b a enter', function() {
    append('konami code!');
});
Mousetrap.bind('* a', function() { append('select all')});
Mousetrap.bind('* n', function() { append('select none')});

Mousetrap.bind('c r a i g', function() { append('craig'); });

// combination and sequence together!
Mousetrap.bind('shift+c r a i g', function() { append('capital craig'); });
